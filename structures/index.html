    Referral Job¶

    These are jobs posted by Referral partners. Seekers will be redirected to the Partner website for application submission.

    id: 
    type: string 
    auto_generated: true
partner_name: 
    type: string
    derived: FK to referral partner
job_title: 
    type: string
    required: true
job_company: 
    type: string
    required: true
job_location: 
    type: string
    required: true
job_description: 
    type: string
    addl: markdown_supported
date_of_posting: 
    type: date
    required: true
monthly_salary: 
    type: number
    required: true 
mode_of_work: 
    type: enum # enum_mode_of_work
    required: true
experience_required: # in years
    type:number

    type: url
